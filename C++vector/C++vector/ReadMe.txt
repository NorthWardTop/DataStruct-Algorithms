// C++vector.cpp : 定义控制台应用程序的入口点。
//

#include "stdafx.h"

#include <vector>
#include <iostream>
#include "CMyVector .h"
using std::vector;

int _tmain(int argc, _TCHAR* argv[])
{
	vector<int> a;//a是动态数组
	for (int i = 0; i < 10; ++i)//一共13个数的内存
	{
		a.push_back(i+1);
	}
	//作业
	a.reserve(100);
	//扩大最大容量,参数为容量大小,提前定义大小(sizeof(T)*参数),不能变小
	vector<int> b(5);//带参(容器大小)构造
	vector<int> c(5, 5);//(大小,元素值)
	c.empty();//判断空
	c.assign(4, 43);//重置容器,将4个43装入容器(容器大小为会变为4,最大大小不变
	*c.swap(b);//a,b交换
	*swap(b, c);//bc交换(友元)
	c.at(2);//返回下标为2的元素值,超范围抛异常
	*c[2];//同上,不检查下标范围,重载[]
	c.front();//返回第一个元素,不检查是否存在
	c.back();//返回最后一个元素
	c.pop_back();//删除最后一个元素,但不返回
	c.resize(6);
	/*
		重置大小,将元素数量(len)改为5,不删除数据,仅仅改变可访问长度
			1.如果大于原长度,len增加,多余部分初始化(0)填充
			2.如果小于原长度,len减小,多余数据不清空,仅改变len长度
	*/
	c.resize(15, 30);
	/*
		如果大于大小,多出部分用参数2赋值
		如果小于大小,改变len,不删除以后数据,仅仅改变访问权限
	*/
	c.clear();//删除所有元素,将容器清空
 	return 0;
}
